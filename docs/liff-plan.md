## LIFF Initialization

```typescript
// src/hooks/useLiff.ts
import liff from '@line/liff';
import { useEffect, useState } from 'react';

export function useLiff() {
  const [isReady, setIsReady] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const [profile, setProfile] = useState<liff.Profile | null>(null);

  useEffect(() => {
    const initLiff = async () => {
      try {
        await liff.init({ liffId: process.env.NEXT_PUBLIC_LIFF_ID! });

        if (!liff.isLoggedIn()) {
          liff.login();
          return;
        }

        const userProfile = await liff.getProfile();
        setProfile(userProfile);
        setIsReady(true);
      } catch (err) {
        setError(err as Error);
      }
    };

    initLiff();
  }, []);

  return { isReady, error, profile, liff };
}
```

// Backend: Validate LIFF access token
async function validateLiffToken(accessToken: string) {
  const response = await fetch('https://api.line.me/oauth2/v2.1/verify', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: `access_token=${accessToken}`,
  });

  if (!response.ok) {
    throw new Error('Invalid LIFF token');
  }

  return response.json();
}
